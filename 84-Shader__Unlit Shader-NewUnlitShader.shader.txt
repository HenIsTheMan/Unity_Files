Shader "MyUnlitShaders/#NAME#" {
    Properties {
        _MainTex("Texture", 2D) = "white" {}
    }

    SubShader {
        Tags {
            "RenderType" = "Opaque"
        }

        LOD 100

        Pass {
            CGPROGRAM

            #pragma vertex VertexMain
            #pragma fragment FragMain

            #include "UnityCG.cginc"

            struct VertexData {
                float4 pos: POSITION;
                float2 texCoords: TEXCOORD0;
            };

            struct FragData {
                float4 pos: SV_POSITION;
                float2 texCoords: TEXCOORD0;
            };

            sampler2D _MainTex;
            float4 _MainTex_ST;

            FragData VertexMain(VertexData vertexData) {
                FragData fragData;

                fragData.pos = UnityObjectToClipPos(vertexData.pos);
                fragData.texCoords = TRANSFORM_TEX(vertexData.texCoords, _MainTex);

                return fragData;
            }

            fixed4 FragMain(FragData fragData): SV_Target {
                fixed4 color = tex2D(_MainTex, fragData.texCoords);

                return color;
            }

            ENDCG
        }
    }
}
